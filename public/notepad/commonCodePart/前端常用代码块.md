### iconfont 提取所有 icon-class

```js
var list = []
$('.font-class .code-name').each((i, item) => {
  let icon = $(item).text().trim().slice(1)
  let name = $(item).prev().text().trim()
  list.push({ name, icon })
})
console.log(JSON.stringify(list))
```
